% \AnforderungDesign{Pr√§fix}{Headline}{Text}{Label}{Counter}
\newcommand{\AnforderungDesign}[5]{
    \noindent\textbf{#1-#5 ~#2}\storelabel{#1-#5 ~#2}\label{Anf:#4} \\
    \noindent\rule[5mm]{\textwidth}{1pt} \\
    \vspace{-15mm} 
        
    \noindent {#3}\nolinebreak
    \vspace{5mm}
}
